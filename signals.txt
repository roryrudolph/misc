#############
#  SIGNALS  #
#############

#######################
# Program Error Signals
#
SIGABRT		Abort
SIGBUS		Bus error
SIGEMT		Emulator trap
SIGFPE		Floating-point exception (or any arithmetic exception)
SIGIOT		Generated by the PDP-11 "iot" instrcution (typically alias for 
			SIGABRT)
SIGSEGV		Segmentation violation
SIGSYS		Bad system call
SIGTRAP		Generally used by debuggers

#####################
# Termination Signals
#
SIGHUP		Hang-up
SIGINT		Program interrupt (Ctrl + c)
SIGKILL		Kill (kill -9 PID)
SIGQUIT		Quit (Ctrl + \)
SIGTERM		Terminate (kill PID)

###############
# Alarm Signals
#
SIGALRM		Generally indicates expiration of a timer measuring the real/clock
			time
SIGPROF		Used to implement code profiling facilities
SIGVTALRM	Virtual time alarm, indicates expiration of a timer that measures
			CPU time

##########################
# Asynchronous I/O Signals
#
SIGIO		Sent when a file descriptor is ready to perform I/O
SIGPOLL		System V signal name, like SIGIO, only used for compatibility 
SIGURG		Urgent request on a particular socket

#####################
# Job Control Signals
#
SIGCHLD		Sent to the parent when a child process terminates
SIGCLD 		Obsolete name for SIGCHLD		
SIGCONT		Sent to a process to make it continue
SIGSTOP		Stops a process, cannot be handled
SIGTSTP		Interactive SIGSTOP, can be handled
SIGTTIN		Sent when a process in a background job tries to read input from
			the terminal
SIGTTOU		Sent when a process in a background job tries to write to a
			terminal

#########################
# Operation Error Signals
#
SIGPIPE		Broken pipe
SIGLOST		Resource lost
SIGXCPU		CPU time limit exceeded
SIGXFSZ		File size limit exceeded

#######################
# Miscellaneous Signals
#
SIGUSR1		User-defined signal, useful for interprocess communciation
SIGUSR2		User-defined signal, useful for interprocess communication
SIGWINCH	Window size change
SIGINFO		Information request
